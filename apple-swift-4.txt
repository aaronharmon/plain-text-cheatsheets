NUMERIC TYPES

var x: Int         // mutable integer variable value with explicit type
var x: Int = 0     // set to initial value
var x = 0          // inferred type
let x = 0          // imutable variable

var y: float        // mutable single floating point variable value with explicit type
var y: float = 0.0  // set to initial value
var y = 0.0         // inferred type
let y = 0.0         // imutable variable

var z: Double = 0.0  // double floating point

var z: UInt = 0      // unsigned integer (larger range, positive only)
